SimpleSlides

A very very simple slideshow, with minimal functionality and easy to integrate.

Example 2.1: Navigation bar - Default

Using the plugin's default DOM and CSS elements.

Slide 1
Slide 2
Slide 3
<div class="simpleslides">
  <div class="simpleslides-item">Slide 1</div>
  <div class="simpleslides-item">Slide 2</div>
  <div class="simpleslides-item">Slide 3</div>
  <div class="simpleslides-nav">
    <div class="simpleslides-nav-item"></div>
    <div class="simpleslides-nav-item"></div>
    <div class="simpleslides-nav-item"></div>
  </div>
</div>

var example = new SimpleSlides({
  showNavigation: true
});
example.start();

Example 2.2: Navigation bar - Custom

Using custom DOM and CSS elements.

Slide 1
Slide 2
Slide 3
Slide 4
<div class="example" id="example">
  <div class="example-item">Slide 1</div>
  <div class="example-item">Slide 2</div>
  <div class="example-item">Slide 3</div>
  <div class="example-item">Slide 4</div>
  <div class="example-nav">
    <div class="example-nav-item"></div>
    <div class="example-nav-item"></div>
    <div class="example-nav-item"></div>
    <div class="example-nav-item"></div>
  </div>
</div>

var example2 = new SimpleSlides({
  parentNode: '#example',
  childrenNode: '.example-item',
  navigationNode: '.example-nav',
  navigationNodeChildren: '.example-nav-item',
  showNavigation: true
});
example2.start();
