var example = new SimpleSlides({
  parentNode: '#example',
  childrenNode: '.example-item',
  navigationNode: '.example-nav',
  navigationNodeChildren: '.example-nav-item',
  slides: 4,
  slide: 3,
  autoplay: false,
  timeout: 1000,
  transitionTime: 100,
  transitionEasing: 'swing',
  showNavigation: true,
  activeClass: 'customActive'
});
example.start();
