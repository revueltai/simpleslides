var example = new SimpleSlides({
  showNavigation: true,
  pauseOnHover: true
});

example.on('start', function(event) {
  console.log('Called event start');
});
example.on('pause', function(event) {
  console.log('Called event pause');
});
example.on('stop', function(event) {
  console.log('Called event stop');
});
example.on('change', function(event) {
  console.log('Called event change');
});

example.start();
