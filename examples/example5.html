var example = new SimpleSlides({
  slide: 4,
  timeout: 500,
  showNavigation: true,
  onStart: function() {
    example.goToNextSlide();
    console.log('Current slide is: ' + example.getCurrentSlide());
  },
  onChange: function() {
    console.log('Current slide is: ' + example.getCurrentSlide() + '. Previous slide is: ' + example.getPreviousSlide() + '. Next slide is: ' + example.getNextSlide());
  }
});
example.start();
