Example 2: Navigation bar

Custom CSS and options to show the navigation bar.

Slide 2
Slide 3

var example2 = new SimpleSlides({

  childrenNode: '.example-item',
  navigationNode: '.example-nav',

});
example2.start();

Example 3: Different options

Playing around with most configuration options.

Slide 1
Slide 2
Slide 3
Slide 4

var example3 = new SimpleSlides({
  parentNode: '#example3',
  childrenNode: '.example-item',
  navigationNode: '.example-nav',
  navigationNodeChildren: '.example-nav-item',
  slides: 4,
  slide: 2,
  autoplay: false,
  timeout: 1000,
  transitionTime: 3000,
  transitionEasing: 'swing',
  showNavigation: true,
  activeClass: 'customActive'
});
example3.start();

Example 4: Events

Supported Events.

Slide 1
Slide 2
Slide 3

var example4 = new SimpleSlides({
  parentNode: '#example4',
  childrenNode: '.example-item',
  navigationNode: '.example-nav',
  navigationNodeChildren: '.example-nav-item',
  showNavigation: true,
  pauseOnHover: true
});
example4.on('start', function(event) {
  console.log('Slideshow start');
});
example4.on('pause', function(event) {
  console.log('Slideshow paused');
});
example4.on('stop', function(event) {
  console.log('Slideshow stop');
});
example4.on('change', function(event) {
  console.log('Slideshow change');
});
